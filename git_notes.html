1. install git
sudo apt install git

2. install git cli
type -p curl >/dev/null || (sudo apt update && sudo apt install curl -y)
curl -fsSL https://cli.github.com/packages/githubcli-archive-keyring.gpg | sudo dd of=/usr/share/keyrings/githubcli-archive-keyring.gpg \
&& sudo chmod go+r /usr/share/keyrings/githubcli-archive-keyring.gpg \
&& echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/githubcli-archive-keyring.gpg] https://cli.github.com/packages stable main" | sudo tee /etc/apt/sources.list.d/github-cli.list > /dev/null \
&& sudo apt update \
&& sudo apt install gh -y

3. authenticate to github
gh auth login
  
4. initialize git 
git init

5. syncronise local folder with server
git remote add origin https://github.com/wtsu/wtsu.github.io.git

6. push local folder to server
git pull origin main

7. create branch 
git branch notes

8. checkout branch 
git checkout notes

9. See if any files need staging
git status

10. add file to staging 
git add git_notes.txt

11. if you want to add all files to staging
git add --all

11. commit changes
git commit -m 'seeding git notes'

12. push to server
git push origin notes
